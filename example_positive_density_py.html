def loadMatrixFromOperation(numGridPoints, op):
alpha = DataVector(numGridPoints)
alpha.setAll(0.0)
result = DataVector(numGridPoints)
ans = np.ndarray((numGridPoints, numGridPoints))
for i in range(numGridPoints):
alpha[i] = 1.0
op.mult(alpha, result)
ans[:, i] = result.array()
alpha[i] = 0.0
return ans
